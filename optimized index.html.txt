        function changeTheme(color) {
            const colors = {
                blue: '#2563eb',

        // Update live data
        function updateLiveData(type, value) {
            switch(type) {
                case 'users':
                    document.getElementById('liveUsers').textContent = parseInt(value).toLocaleString();
                    break;
                case 'models':
                    document.getElementById('totalModels').textContent = value + 'K';
                    break;
                case 'accuracy':
                    document.getElementById('aiAccuracy').textContent = value + '%';
                    break;
            }
        }
        
        // Analyze input
        function analyzeInput() {
            const input = document.getElementById('aiInput').value;
            const parametersGrid = document.getElementById('parametersGrid');
            
            parametersGrid.innerHTML = '';
            
            const parameters = [];
            
            if (input.includes('window')) {
                parameters.push({ name: 'Window Height', value: '4', unit: 'ft' });
                parameters.push({ name: 'Window Width', value: '3', unit: 'ft' });
            }
            
            if (input.includes('door')) {
                parameters.push({ name: 'Door Height', value: '7', unit: 'ft' });
                parameters.push({ name: 'Door Width', value: '3', unit: 'ft' });
            }
            
            if (input.includes('wall')) {
                parameters.push({ name: 'Wall Thickness', value: '12', unit: 'in' });
            }
            
            if (input.includes('floor') || input.includes('story')) {
                parameters.push({ name: 'Floor Height', value: '12', unit: 'ft' });
                parameters.push({ name: 'Floor Count', value: '5', unit: '' });
            }
            
            parameters.forEach(param => {
                const card = document.createElement('div');
                card.className = 'parameter-card';
                card.innerHTML = `
                    <span class="parameter-name">${param.name}</span>
                    <div class="parameter-value">
                        <input type="text" class="parameter-input" value="${param.value}">
                        <span class="parameter-unit">${param.unit}</span>
                    </div>
                `;
                parametersGrid.appendChild(card);
            });
        }
